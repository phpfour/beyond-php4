Syntax Improvements

Array Syntax

// Works on all versions
$games = array('football', 'cricket', 'basketball');
print_r($games);

// Works on 5.4+
$games = ['football', 'cricket', 'basketball'];
print_r($games);

Instant access to new objects

// Old way
$postModel = new Post();
$latestPosts = $postModel->getLatest();

// Works on 5.4+
$latestPosts = (new Post())->getLatest();

Instant access to returned arrays

// Old way
$postModel = new Post();
$latestPosts = $postModel->getLatest();
$firstPost = $latestPosts[0];

// Works on 5.4+
$firstPost = (new Post())->getLatest()[0];

New Ternary Operator

// Old way
$action = $_POST['action'] ? $_POST['action'] : 'default_action';

// Works on 5.3+
$action = $_POST['action'] ?: 'default_action';

Namespaces

Prevents naming conflict

Keep things organized

Before namespaces

// modules/blog/Post.php
class Post { ... }

// modules/news/Post.php
class Post { ... }

$post = new Post();

Fatal error: Cannot redeclare class Post

After namespaces

// modules/blog/Post.php
namespace Blog;
class Post { ... }

// modules/news/Post.php
namespace News;
class Post { ... }

$post = new Blog\Post();

Closures

Very common in languages like JavaScript

Now extremely popular in PHP

Javascript Usage (jQuery)

$('.hide').click(function(){
    $(this).hide();
});

PHP Usage

// Old way
function is_even($n) {
    return $n % 2 === 0;
}

$numbers = [1, 2, 3, 4, 5];
$evens = array_filter($numbers, "is_even");

// New way
$numbers = [1, 2, 3, 4, 5];
$evens = array_filter($numbers, function($n){
    return $n % 2 === 0;
});

Type Hinting

Force methods to only accept certain types of arguments

class Post
{
    protected $author;
    protected $tags = [];
    
    public function __construct(Author $author, array $tags)
    {
        $this->author = $author;
        $this->tags = $tags;
    }
}

Traits

PHP 5.4+

Alternative to multi-class inheritance

Re-usable group of methods

trait Taggable
{
    public function getTags()
    {
        return Tag::where('post_id', '=', $this->id)->get();
    }
    
    public function countTags()
    {
        return Tag::where('post_id', '=', $this->id)->count();
    }
}

trait Categorizeable
{
    public function getCategories()
    {
        return Category::where('post_id', '=', $this->id)->get();
    }
    
    public function countCategories()
    {
        return Category::where('post_id', '=', $this->id)->count();
    }
}

class Post
{
    use Taggable, Categorizeable;
    
    protected $id;
    
    public function __construct($id = null)
    {
        $this->id = $id;
    }
}

$post = new Post(1);
$tags = $post->getTags();
$categories = $post->getCategories();
